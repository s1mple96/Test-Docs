public String notify(@PathVariable("payChannelCode") String payChannelCode, 
                     HttpServletResponse response, HttpServletRequest request) {
    logger.info("进入账户出金模块异步回调方法，payChannelCode：{}", payChannelCode);

    Map<String, Object> parameterMap = getParameterMap();

    AccountTransNotifyVerifyParam param = new AccountTransNotifyVerifyParam();

        PayChannelCode payChannelCodeObj = payChannelCodeService.getWithdrawChannelByBankChannelCode(payChannelCode);

#### 2. 核心验签逻辑
```java
// LingYueApi.java:335-352
public static JSONObject notifyVerify(AccountTransNotifyVerifyParam transParam, PayChannelCode channelCode) throws Exception {
    // 获取灵跃平台公钥（用于验签）
    String publicKey = channelCode.getMerchantPublickey();
    
    // 获取回调参数
    Map<String, Object> content = transParam.getParamMap();
    Assert.notNull(content, "灵鹊云灵活用工回调数据为空！");  // ✅ 已修复
    
    // 转换参数格式
    Map<String, String> param = new HashMap<>();
    for(String key : content.keySet()) {
        param.put(key, null==content.get(key)?"":content.get(key).toString());
    }
    
    // ⚠️ 关键：RSA2签名验证
    boolean flag = SecSignature.rsaCheckV2(param, publicKey, 
                                           SecGatewayConstants.CHARSET_UTF8,
                                           SecGatewayConstants.SIGN_TYPE_RSA2);
    
    // 记录验签日志到MongoDB
    BankLogEventSend.saveMongoLog(JSON.toJSONString(content), JSON.toJSONString(param), 
                                   "", "0", "0",
                                   BankLogBizTypeEnum.WITHDRAW_NOTIFY, 
                                   BankCollectionNameEnum.LINGYUE);
    
    // ⚠️ 关键：验签失败直接抛异常，阻止后续处理
    Assert.isTrue(flag, "灵跃用工回调验签失败！");
    
    // 验签成功，解析业务数据
    JSONObject resultObject = JSON.parseObject(param.get(SecGatewayConstants.BIZ_CONTENT_KEY));
    return resultObject;
}
```
